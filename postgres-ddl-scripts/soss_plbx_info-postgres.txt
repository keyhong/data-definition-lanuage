CREATE TABLE PLBX_INFO (
	plbx_nm VARCHAR(10) primary key, 
	plbx_adrs VARCHAR(50) NULL,
	plst_nm VARCHAR(10) NULL,
	gu_cd VARCHAR(5) NULL,
	la DECIMAL(18,8) NULL,
	lo DECIMAL(18,8) NULL
);

COMMENT ON TABLE ADMD_PLBX_MAPN IS '지구대정보';

COMMENT ON COLUMN ADMD_PLBX_MAPN.plbx_nm IS '지구대명@pk';
COMMENT ON COLUMN ADMD_PLBX_MAPN.plbx_adrs IS '지구대주소';
COMMENT ON COLUMN ADMD_PLBX_MAPN.plst_nm IS '경찰서명';
COMMENT ON COLUMN ADMD_PLBX_MAPN.gu_cd IS '구코드';
COMMENT ON COLUMN ADMD_PLBX_MAPN.la IS '위도';
COMMENT ON COLUMN ADMD_PLBX_MAPN.lo IS '경도';
